from cassandra.cluster import Cluster cluster = Cluster(['127.0.0.1']) session = cluster.connect('killrvideo') rows = session.execute( """SELECT * FROM latest_videos WHERE yyyymmdd = %s ORDER BY added_date DESC LIMIT 10""", ['2024-01-15'] ) for row in rows: print(row.title)
